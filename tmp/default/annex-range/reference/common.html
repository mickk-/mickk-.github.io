template<ForwardableType Functor, MoveConstructible Rng>
constexpr auto operator()(Functor &&functor, Rng rng) const¶
template<StorableType PeekFunctor, StorableType Functor, MoveConstructible Rng>
constexpr auto operator()(PeekFunctor &&peek_functor, Functor &&functor, Rng rng) const¶

Equivalent to:

// first overload
range::flatten(range::map(std::forward<Functor>(functor), std::move(rng)))

// second overload
range::flatten(range::map(std::forward<PeekFunctor>(peek_functor), std::forward<Functor>(functor), std::move(rng)))
