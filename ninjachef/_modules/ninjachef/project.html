[docs]class Files(object):
    def __init__(self, *paths, pattern=None, patterns=None):
        if pattern is None and patterns is None:
            self.paths = set(map(PurePath, paths))
        else:
            ps = set()

            patterns = patterns or []
            if pattern:
                patterns.append(pattern)

            for pattern in patterns:
                for root in paths:
                    ps.update(Path(root).rglob(pattern))

            self.paths = ps

    def __repr__(self):
        return '{}({})'.format(type(self).__name__, ', '.join(map(repr, self.paths)))

    """The collected ninjachef-information of a project.

    Attributes:
        path (PurePath): The root path of the project, which serves as a base
            for :class:`Ingredient` paths. It can be relative or absolute,
            although some recipes might require an absolute path.
        descrs (Dict): The mapping of names to :class:`Ingredient` s as
            described by the project's ingredient file.

    Args:
        path (PurePath): Root path of the project.
        descrs (Dict): Mapping of names to :class:`Ingredient` s.
    """

    def __init__(self, path, descrs):
        self.path = path
        self.descrs = descrs

    def __repr__(self):
        descr = '{0}(path={1.path}, descrs={1.descrs})'
        return descr.format(type(self).__name__, self)

[docs]    def read(cls, path, file='ingredients.py'):
        """Analyse a project to collect :class:`Ingredient` information.

        Args:
            path (PurePath): The root path of the project. An ingredient file
                within that path will be looked up (see ``file``) and is assumed
                to contain a description of the project (see
                :mod:`ninjachef.cookbook`).

        Keyword Args:
            file (str): The name of the ingredient file (default:
                ``ingredients.py``).

        Raises:
            OSError: if reading the ingredient file fails.
            ImportError: if importing the ingredient file fails.

        Returns:
            a :class:`Project` object describing the project.
        """
        with cd(path):
            # load_module is deprecated in 3.4 but there is no suitable alternative yet
            loader = importlib.machinery.SourceFileLoader("ingredients", file)
            module = loader.load_module()
            descrs = { k: v for k, v in module.__dict__.items() if isinstance(v, Ingredient) }
            return cls(path, descrs)
