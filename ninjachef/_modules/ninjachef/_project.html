[docs]class Deps(object):
    """A list of dependencies.

    Attributes:
        deps (List[Union[str, Tuple[str, ...]])
        local_deps (List[Ingredient])

    Args:
        *deps (Tuple[Union[str, Tuple[str, ...], ...]): A tuple of external
            dependencies. It will be up to the recipe to interpret and take the
            appropriate precautions for each dependency.

            Each individual dependency can take the form of a :class:`.str`
            (e.g. ``'boost_system'``) or a `Tuple[str, ...]` (e.g. ``('foo',
            'bar')``). With these examples, a recipe might choose to use the
            system-installed Boost.System library and the `bar` component of a
            ninjachef-aware `foo` project.

    Keyword Args:
        local (Sequence[Ingredient]): A sequence of internal dependencies. Those
            must be other :class:`Ingredients <.Ingredient>` defined in the same
            project. This typically happens for testsuites::

                self = Binary(…)
                test_files = Files(…)
                test = Binary(self.files | test_files,
                              main=…,
                              deps=Deps(local=[self]))
    """

    def __init__(self, *deps, local=None):
        self.deps       = list(deps)
        self.local_deps = list(local) if local else []

    def __repr__(self):
        message = '{{cls}}({args})'.format(args=', '.join(filter(None, ['{deps}' if self.deps else '', 'local={self.local_deps}' if self.local_deps else ''])))
        return message.format(cls=type(self).__name__, self=self, deps=', '.join(map(repr, self.deps)))


[docs]    def __or__(self, other):
        """Merge two lists of dependencies.

        Examples:
            >>> d = Deps('a') | Deps('b') | Deps(('a', 'b'))
            >>> list(map(str, d.deps))
            ['a', 'b', "('a', 'b')"]
        """

        r = Deps()
        r.deps = self.deps + other.deps
        r.local_deps = self.local_deps + other.local_deps
        return r



[docs]class Files(object):
    """A set of paths to files.

    A :class:`.Files` object can be created from either :class:`.str` paths (but
    not separate path fragments) or :class:`.PurePath` objects.

    Attribute:
        paths (Set[PurePath])

    Args:
        *paths (Tuple[Union[str, PurePath], ...]): The paths to the files to
            refer to; or the paths to glob from (see keyword arguments
            `pattern`, `patterns`).

    Keyword Args:
        pattern (Optional[str]): A glob-style pattern which will be used with
            the specified `paths` to collect all matching files. If neither
            `pattern` nor `patterns` is provided, no globbing (or any
            file system operation) will be performed and the specified `paths`
            will be stored directly.
        patterns (Optional[Sequence[str]]): A sequence of glob-style patterns to
            use for matching, see `pattern`.

    Examples:
        >>> # Use the specified paths directly, no file system operation
        >>> fs = Files('a', os.path.join('b', 'c'))
        >>> len(fs.paths)
        2
        >>> # Will walk down a and b/c to collect all matching files
        >>> Files('a', PurePath('b', 'c'), patterns=['*.cxx', '*.cpp'])
        Files(...)
    """

    def __init__(self, *paths, pattern=None, patterns=None):
        if pattern is None and patterns is None:
            self.paths = set(map(PurePath, paths))
        else:
            ps = set()

            patterns = list(patterns or [])
            if pattern:
                patterns.append(pattern)

            for pattern in patterns:
                for root in paths:
                    ps.update(Path(root).rglob(pattern))

            self.paths = ps

    def __repr__(self):
        return '{cls}({paths})'.format(cls=type(self).__name__, paths=', '.join(map(repr, self.paths)))


[docs]    def __or__(self, other):
        """Merge two sets of files.

        Examples:
            >>> f = Files('a') | Files('b') | Files(PurePath('a'))
            >>> sorted(map(str, f.paths))
            ['a', 'b']
        """

        r = Files()
        r.paths = self.paths | other.paths
        return r



[docs]class Ingredient(object):
    """The base class for all the sorts of ingredients that can be used in
    describing a project.

    This is emphatically not an Abstract Base Class, and it is not meant to be
    extended or derived from. This base class is used internally and is
    documented for exposition purposes, in particular for recipe writers.

    The ninjachef ingredients are intended to form a closed world of relatively
    broad categories of things. In return, it makes it possible to define new
    recipes (which are consumers of :class:`Projects <.Project>`, and hence
    ingredients), even long after project descriptions have been written.
    """

[docs]    def split(cont, by=None, cls=None):
        """Split a container of :class:`Ingredients <.Ingredient>` into a
        triple of :class:`.ProjectDeps`, :class:`.Headers`, and :class:`.Binary`
        containers respectively.

        Args:
            cont: The container to split. Must contain only :class:`Ingredients
                <.Ingredient>` unless the `by` argument is specified. Can be an
                iterable if `cls` is specified, but not an iterator.

        Returns:
            A triple of containers, each discriminating against a specific kind
            of :class:`.Ingredient`. In order: :class:`.ProjectDeps`,
            :class:`.Headers`, and :class:`.Binary`.

            The containers are the same as `cont` (e.g. a triple of :class:`sets
            <.set>` for a :class:`.set` argument), unless `cls` is specified.

        Keyword Args:
            by (Optional[Union[str, Callable[..., Ingredient]]]): A projection
                to single out an :class:`.Ingredient` out of a container
                element, which will be used for partitioning. If not specified,
                equivalent to using an identity function.

                If a :class:`.str` is specified (e.g. ``'foo'``), equivalent to
                passing ``by=lambda x: getattr(x, 'foo')``. Otherwise, a
                callable is assumed and will be used on each element of the
                container.

                E.g. specifing ``by=lambda x: x[1]`` allows a container of
                ``(name, ingr)`` pairs to be split according to the second
                element of each pair. Note that the returned containers will
                still contain ``(name, ingr)`` pairs (of appropriate type).
                Compare to splitting a ``map(lambda x: x[1], cont)``, which
                produces a triple of :class:`.Ingredient` containers.

            cls (Optional[Type]): Explicit container type to use in the returned
                triple. This makes it possible to pass in an iterable instead of
                an actual container (e.g. a :class:`.dict_items`, when splitting
                a :class:`.dict`).

        Example:
            >>> e = Files()
            >>> ingrs = [Headers(e), ProjectDeps(), Headers(e), Binary(e), Binary(e), Binary(e)]
            >>> ds, hs, bs = Ingredient.split(ingrs)
            >>> len(ds)
            1
            >>> len(hs)
            2
            >>> len(bs)
            3
        """

        if isinstance(by, str):
            attr = by
            by = lambda x: getattr(x, attr)
        else:
            by = by or (lambda x: x)

        cls = cls or type(cont)
        def inst(base):
            return lambda x: isinstance(by(x), base)
        return cls(filter(inst(ProjectDeps), cont)), cls(filter(inst(Headers), cont)), cls(filter(inst(Binary), cont))

    """Project-wide external dependencies.

    Attributes:
        project_deps (Deps): External dependencies, meaning that
            ``project_deps.local_deps`` is empty.

    Args:
        *deps (Tuple[Union[str, Tuple[str, ...]], ...]): A tuple of external
            dependencies. See :class:`.Deps`.

    def __init__(self, *deps):
        self.project_deps = Deps(*deps)

    def __repr__(self):
        return '{cls}({deps})'.format(cls=type(self).__name__, deps=', '.join(map(repr, self.project_deps.deps)))

        """Merge two lists of external dependencies.

[docs]class Headers(Ingredient):
    """Installable headers.

    This can be used by library projects (or library components within a
    project) to specify that a particular set of headers is installable, e.g.
    for development purposes.

    This should *not* be used to describe how e.g. C++ ``#include`` directives
    should be resolved, see :class:`.Binary` (in particular the
    `includes` keyword argument) instead for that purpose.

    Attributes:
        files (Files)
        roots (Set[PurePath])

    Args:
        files (Files): The set of headers that should be installed.

    Keyword Args:
        roots (Sequence[Union[str, PurePath]): a sequence of paths serving as
            candidate roots for the `files`. This allows the recipe to
            eventually install headers at the correct location. E.g. given the
            following::

                hs = Headers(Files(path('include', 'location_0', 'foo.hpp'),
                                   path('include', 'location_a', 'location_b', 'bar', 'qux.hpp')),
                             roots=[path('include', 'location_0'),
                                    path('include', 'location_a', 'location_b')])

            then a recipe will know to install both ``foo.hpp`` and
            ``bar/qux.hpp`` side-by-side at the same location (where exactly it
            will do so depends on the precise recipe).
    """

    def __init__(self, files, roots=None):
        self.files = files
        self.roots = set(map(PurePath, roots or set()))

    def __repr__(self):
        message = '{cls}(files={self.files}, roots={self.roots})'
        return message.format(cls=type(self).__name__, self=self)



[docs]class Binary(Ingredient):
    """Binary file.

    Represents a given set of files that together can be compiled into a binary
    file. If a `main` file is specified (see keyword argument `main`), the
    produced binary is assumed to be an executable. Otherwise, it will be
    assumed to be a library. A recipe may choose to override this behavior.

    Attributes:
        files (Files): The files other than `main`.
        main (Optional[PurePath]): The file containing the appropriate ``main``
            function, if any.
        includes (Set[PurePath])
        deps (Deps)
        defines (Set[str]): A set of C or C++ defines that should be used when
            building this binary. This should be reserved for defines that are
            unconditionally set. Build-specific flags (e.g. ``NDEBUG`` for a
            release build) should be specified via the flavor.

    Args:
        files (Files): The set of files that result in a binary. If the keyword
            argument `main` is provided, its associated file will be ignored
            here. This is convenient for executable tests which might want to
            run their own ``main``::

                self = Binary(Files(…),
                              main=path('source', 'application_main.cpp'))
                test = Binary(self.files | Files(…),
                              main=path('test', 'test_main.cpp'))

    Keyword Args:
        main (Optional[Union[str, PurePath]]): An optional path to the file that
            contains the appropriate ``main`` function, if relevant.
        includes (Sequence[Union[str, PurePath]]): A sequence of paths
            (relative to the project root) which are needed to properly resolve
            C++ ``#include`` directives when it comes to the project-specific
            headers.
        deps (Deps): Binary-specific dependencies. If a dependency applies to
            all binaries in a given project, consider using a separate
            :class:`.ProjectDeps` component instead.
    """

    def __init__(self, files, main=None, includes=None, deps=None):
        main = PurePath(main) if main else None

        self.files    = Files(*(file for file in files.paths if file != main))
        self.main     = main
        self.includes = set(map(PurePath, includes or []))
        self.deps     = copy(deps) if deps else Deps()
        self.defines  = set()

    def __repr__(self):
        message = '{cls}({self.files}, main={self.main}, includes={self.includes}, deps={self.deps})'
        return message.format(cls=type(self).__name__, self=self)

    """Error when analyzing an ingredient file.

    Raised when loading an ingredient file goes wrong. Should be chained with
    the actual exception that was raised during the import.

    Attributes:
        path (Path): The path to the responsible ingredient file.

    Args:
        path (Union[str, Path]): Should be absolute for user-friendliness.

    def __init__(self, path):
        super().__init__('error while importing ingredient file {path!r}'.format(path=str(path)))
        self.path = PurePath(path)



[docs]class Project(object):
    """The collected ninjachef-information of a project.

    Present a :mod:`project description <.cookbook>` in Python form.

    Attributes:
        path (PurePath): The root path of the project, which serves as a base
            for :class:`.Ingredient` paths. It can be relative or absolute.
        deps (Dict[str, Deps]): The mapping of names to :class:`.Deps`.
        headers (Dict[str, Headers]): The mapping of names to :class:`.Headers`.
        binaries (Dict[str, Binary]): The mapping of names to :class:`Binaries
            <.Binary>`.
        descr (Iterable[(str, Ingredient)]): The mapping of names to
            :class:`Ingredients <.Ingredient>`. Equivalent to an iterable
            containing all the entries of `deps`, `headers`, `binaries`.

    Args:
        path (Union[str, PurePath]): Root path of the project.
        descr (Mapping[str, Ingredient]): Project components, in the form of a
            mapping of names to :class:`Ingredients <.Ingredient>`.

    Raises:
        ValueError: if `descr` does not map names to :class:`Ingredients
            <.Ingredient>`
    """

    def __init__(self, path, descr):
        deps     = dict()
        headers  = dict()
        binaries = dict()

        partitioning = [(ProjectDeps, deps), (Headers, headers), (Binary, binaries)]
        for name, comp in descr.items():
            for cls, cont in partitioning:
                if isinstance(comp, cls):
                    cont[name] = comp
                    break
            else:
                raise ValueError("descr is not a mapping of names to ingredients: {descr}".format(descr=descr))

        self.path     = PurePath(path)
        self.deps     = deps
        self.headers  = headers
        self.binaries = binaries

    def __repr__(self):
        message = '{cls}(path={self.path}, descr={descr})'
        return message.format(cls=type(self).__name__, self=self, descr=dict(self.descr))

    @property
    def descr(self):
        return chain(self.deps.items(), self.headers.items(), self.binaries.items())

[docs]    def read(cls, path, file='ingredients.py'):
        """Analyze a project to collect :class:`.Ingredient` information.

        Args:
            path (Union[str, PurePath]): The root path of the project. An
                ingredient file within that path will be looked up (see the
                `file` keyword argument) and is assumed to contain a description
                of the project (see :mod:`ninjachef.cookbook`).

        Keyword Args:
            file (str): The name of the ingredient file.

        Raises:
            OSError: if reading the ingredient file fails.
            IngredientError: if importing the ingredient file otherwise fails.

        Returns:
            Project: A description of the project.
        """

        with cd(path):
            # load_module is deprecated in 3.4 but there is no suitable alternative yet
            loader = importlib.machinery.SourceFileLoader(PurePath(file).stem, file)
            try:
                module = loader.load_module()
            except OSError:
                raise
            except Exception as exc:
                raise IngredientError(Path().resolve() / file) from exc
            descr = {k: v for k, v in module.__dict__.items() if isinstance(v, Ingredient)}
            return cls(path, descr)
